ImmutableList<String> donutsInPopularityOrder = this.orders
        .aggregateBy(
                Lists.immutable.of(sum("Quantity")),
                Lists.immutable.of("Donut"))
        .sortBy(Lists.immutable.of("Quantity", "Donut"))
        .getStringColumn("Donut")
        .toList();
